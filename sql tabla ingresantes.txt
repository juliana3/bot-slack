

CREATE TABLE public.new_entrants (
	id serial4 NOT NULL,
	first_name varchar(100) NOT NULL,
	lastname varchar(100) NOT NULL,
	dni varchar(20) NOT NULL,
	email varchar(100) NOT NULL,
	phone_number varchar(20) NOT NULL,
	address varchar(255) NOT NULL,
	locality varchar(100) NOT NULL,
	date_of_birth varchar(20) NOT NULL,
	health_insurance varchar(100) NOT NULL,
	afip_code varchar(50) NULL,
	cbu varchar(50) NULL,
	alias varchar(100) NULL,
	cuil varchar(50) NULL,
	national_bank varchar(100) NULL,
	national_account_number varchar(100) NULL,
	bank_name varchar(100) NULL,
	bank_address varchar(255) NULL,
	swift_code varchar(50) NULL,
	account_holder varchar(100) NULL,
	account_number varchar(50) NULL,
	routing_number varchar(50) NULL,
	account_type varchar(50) NULL,
	zip_code varchar(20) NULL,
	type_of_contract varchar(50) NOT NULL,
	dni_front text NOT NULL,
	dni_back text NOT NULL,
	onboarding_status varchar(50) DEFAULT 'Pendiente'::character varying NULL,
	id_pf varchar(50) NULL,
	pdf_status varchar(50) DEFAULT 'Pendiente'::character varying NULL,
	id_drive_folder varchar(50) NULL,
	CONSTRAINT new_entrants_dni_key UNIQUE (dni),
	CONSTRAINT new_entrants_email_key UNIQUE (email),
	CONSTRAINT new_entrants_pkey PRIMARY KEY (id)
);
